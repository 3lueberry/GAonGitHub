\d : 0-9
\D : not 0-9 : letters
\s : whitespace
\S : not whitespace
'  : any char : not line breaks
+  : 1 or more
*  : 0 or more
?  : 1 or 0
{3} : exactly 3
{2,4} : min 2, max 4
{3, } : min 3, no max
\ \. \* \+ \? \$
^ : start of string
$ : end of string
[abc] : either 1
[abc]defg : adefg / bdefg / cdefg
[^abc] : neither 1
